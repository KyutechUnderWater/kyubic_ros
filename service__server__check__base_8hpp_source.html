64 auto start_time = std::chrono::steady_clock::now();
65 auto timeout_duration = std::chrono::milliseconds(timeout_ms_);
66
67 // 自身の完全修飾名を取得 (例: "/my_ns/checker_node")
68 // これを使って自身をカウントから除外する
69 std::string current_node_name = node->get_fully_qualified_name();

71 while (true) {
72 size_t count = 0;
73
74 // 1. 全ノード名を取得
75 auto node_names = node->get_node_names();
76
77 // 2. 各ノードを巡回
78 for (const auto & full_node_name : node_names) {
79 // --- 自身の除外処理 ---
80 if (full_node_name == current_node_name) {
81 continue;
82 }
83
84 // --- 名前空間とノード名の分離処理 ---
85 std::string target_node_name;
86 std::string target_node_ns;
87
88 size_t last_slash = full_node_name.find_last_of('/');
89 if (last_slash == std::string::npos || last_slash == 0) {
90 target_node_ns = "/";
91 target_node_name = (last_slash == 0) ? full_node_name.substr(1) : full_node_name;
92 } else {
93 target_node_ns = full_node_name.substr(0, last_slash);
94 target_node_name = full_node_name.substr(last_slash + 1);
95 }
96
97 // --- 派生クラスごとのカウント処理 ---
98 if (check_entity_in_node(node, target_node_name, target_node_ns)) {
99 count++;
100 }
101 }
102
103 // 判定
104 if (count >= expected_count_) {
105 status_msg_ = "Found " + std::to_string(count) + " " + get_target_type() + "(s).";
106 return true; // PASS
107 }
108
109 // タイムアウト判定
110 auto elapsed = std::chrono::steady_clock::now() - start_time;
111 if (elapsed > timeout_duration) {
112 status_msg_ = "Timeout. Found " + std::to_string(count) + " " + get_target_type() +
113 "(s) (Expected >= " + std::to_string(expected_count_) + ").";
114 return false; // FAIL
115 }

117 std::this_thread::sleep_for(std::chrono::milliseconds(100));
118 }
